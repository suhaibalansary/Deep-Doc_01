<div id="docsList">
  <h3>المستندات المضافة:</h3>
  <ul id="docsUl"></ul>
</div>

<script>
// تحميل المستندات من localStorage عند فتح الصفحة
window.onload = function() {
  loadDocuments();
}

function addDocument() {
  const fileInput = document.getElementById('fileInput');
  const file = fileInput.files[0];
  if (!file) {
    alert('رجاءً اختر ملف لإضافته.');
    return;
  }

  const reader = new FileReader();
  reader.onload = function(e) {
    const content = e.target.result;
    // نحفظ الملف في localStorage مع اسم المادة "معالجة النصوص"
    let docs = JSON.parse(localStorage.getItem('ma3aljatAlnususDocs')) || [];
    docs.push({ name: file.name, content: content });
    localStorage.setItem('ma3aljatAlnususDocs', JSON.stringify(docs));
    fileInput.value = ""; // نعيد تعيين اختيار الملف
    loadDocuments();
  };
  reader.readAsDataURL(file); // نقرأ الملف كـ base64 حتى نقدر نعرضه أو نستخدمه بعدين
}

function loadDocuments() {
  const docsUl = document.getElementById('docsUl');
  docsUl.innerHTML = '';
  let docs = JSON.parse(localStorage.getItem('ma3aljatAlnususDocs')) || [];
  if (docs.length === 0) {
    docsUl.innerHTML = '<li>لا يوجد مستندات مضافة بعد.</li>';
    return;
  }
  docs.forEach((doc, index) => {
    const li = document.createElement('li');
    li.classList.add('doc-item');
    // رابط لتحميل أو فتح الملف مباشرة
    li.innerHTML = `<a href="${doc.content}" download="${doc.name}">${doc.name}</a> 
                    <button onclick="deleteDocument(${index})">حذف</button>`;
    docsUl.appendChild(li);
  });
}

function deleteDocument(index) {
  let docs = JSON.parse(localStorage.getItem('ma3aljatAlnususDocs')) || [];
  docs.splice(index, 1);
  localStorage.setItem('ma3aljatAlnususDocs', JSON.stringify(docs));
  loadDocuments();
}
</script>
